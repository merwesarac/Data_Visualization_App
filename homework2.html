import pandas as pd
import matplotlib.pyplot as plt
from matplotlib import style
import numpy as np
from matplotlib.ticker import FuncFormatter

plt.style.use('default')

def problem1(x):
    unemployment=pd.read_csv(x)
    plt.figure(figsize=(6, 4))
    
    
    ax = plt.axes()
    ax.set_facecolor('#E2E1E1')
    ax.set_axisbelow(True)
    # draw solid white grid lines
    plt.grid(color='w', linestyle='solid')
    
    x=range(0,122)
    
    # Define a custom formatter function
    def millions_formatter(x, pos):
        return f'{x/1e6:.1f}M'
    
    # Define colors for each variable
    colors = ['3E74AE', '2EAB72', 'C04846', '814EA9', 'D5B670', '61B3D0', '3E74AE']
    
    # Basic stacked area chart.
    plt.stackplot(x,unemployment['16 to 19 months'],
                    unemployment['20 to 24 months'],
                    unemployment['25 to 34 months'],
                    unemployment['35 to 44 months'],
                    unemployment['45 to 54 months'],
                    unemployment['55 to 64 months'],
                    unemployment['65 months and over'],
                    labels=['16 to 19 months','20 to 24 months','25 to 34 months','35 to 44 months','45 to 54 months','55 to 64 months','65 months and over'] ,
                    colors = colors
                 )
    
    # Apply the custom formatter to the y-axis
    formatter = FuncFormatter(millions_formatter)
    plt.gca().yaxis.set_major_formatter(formatter)
    
    plt.legend(loc='upper left',frameon=False,fontsize='6.9')
    plt.xlabel('Month',fontsize='12')
    plt.ylabel('Unemployment',fontsize='12')
    
    ax.spines['top'].set_visible(False)
    ax.spines['bottom'].set_visible(False)
    ax.spines['right'].set_visible(False)
    ax.spines['left'].set_visible(False)

epidemic = pd.read_csv('corona_china_vs_rest.csv')

# Sample data
days = range(0,36)
y1 = epidemic['rest']
y2 = epidemic['china']

# Create the figure and the first y-axis
fig, ax1 = plt.subplots(figsize=(5,5))

# Plot the first line chart with square marker
line1, = ax1.plot(days, y1, color='red', marker='s', markersize=6, label='rest (left-axis)')
ax1.set_xlabel('Days')
ax1.tick_params(axis='y')

# Create the second y-axis
ax2 = ax1.twinx()

# Plot the second line chart with triangle marker
line2, = ax2.plot(days, y2, color='blue', marker='v', markersize=6, label='china (right-axis)')
ax2.tick_params(axis='y')

# Add legends for both lines
lines = [line2, line1]
labels = [line.get_label() for line in lines]
plt.legend(lines, labels,frameon = False)
plt.title('Number of cases')
ax1.spines['top'].set_visible(False)
ax2.spines['top'].set_visible(False)
#save the plot
plt.savefig('china_and_rest.png',bbox_inches = 'tight')

import dash
from dash.dependencies import Input, Output
import plotly.graph_objs as go
import pandas as pd
import plotly.io as pio
from dash import html
from dash import dcc

pio.templates.default = "plotly_white"

# Read the dataset and do manipulations respectively
data = pd.read_csv('covid19_ecdc_28_07_20.csv')
data.dateRep = pd.to_datetime(data.dateRep, format='%d/%m/%Y')
confirmed_by_country = data.pivot(index='countriesAndTerritories', columns='dateRep', values='cases')
deaths_by_country = data.pivot(index='countriesAndTerritories', columns='dateRep', values='deaths')
confirmed_7day_mean = confirmed_by_country.rolling(7, axis=1, min_periods=1).mean()
epidemic = confirmed_7day_mean
epidemic_cumulative = confirmed_by_country.cumsum(axis=1)

# Create Dash app

app = dash.Dash(__name__)

# Define app layout for date and country dropdowns and plot the linecharts
app.layout = html.Div(
    children=[
        html.Div([
            html.Label('Date'),
            dcc.Dropdown(
                id='date-dropdown',
                options=[{'label': date.strftime("%m/%d/%Y"), 'value': date} for date in epidemic.columns[1:]],
                value=epidemic.columns[205],
                clearable=False,
                style={'width': '99px','margin-bottom':'-16px'}
            )
        ], style={'margin-bottom': '20px'}), 
        html.Div([
            html.Label('Country'),
            dcc.Dropdown(
                id='country-dropdown',
                options=[{'label': country, 'value': country} for country in confirmed_by_country.index],
                value='Canada',
                style={'width': '190px','backgroundColor': 'white'}
            )
        ], style={'margin-bottom': '20px'}),
        html.Div('Number of Cases', style={'margin-top': '-16px','font-size':'23px','backgroundColor': 'white'}),
        html.Div(id='cases-text', style={'font-size': '13px', 'white-space': 'normal','margin-top': '6px','backgroundColor': 'white'}),
        html.Div('Number of Deaths', style={'margin-top': '-3px','font-size':'23px','backgroundColor': 'white'}),
        html.Div(id='cases-text-deaths', style={'font-size': '13px', 'white-space': 'normal','margin-top': '2px','backgroundColor': 'white'}),
        #ploting the linecharts
        dcc.Graph(id='line-chart', style={'position':'absolute','top':'3px','left':'230px','width': '500px', 'height': '500px', 'margin': '0 auto','backgroundColor': 'white'}),
        dcc.Graph(id='cumulative-chart', style={'position':'absolute','top':'3px','left':'800px','width': '500px', 'height': '500px', 'margin': '0 auto','backgroundColor': 'white'})
    ],
    style={'width':'600px','height':'600px'}
)

# Define 'number of cases' linechart callbacks
@app.callback(
    Output('line-chart', 'figure'),
    [Input('country-dropdown', 'value')]
)
def update_line_chart(selected_country):
    x = epidemic.columns[1:]
    y = epidemic.loc[selected_country, x].values.flatten()
    
    trace = go.Scatter(
        x=x,
        y=y,
        mode='lines',
        name=selected_country
    )
    
    layout = go.Layout(
        xaxis={'title': 'Date','showgrid': True,'gridcolor':'white'},
        yaxis={'title': 'Number of Cases', 'tickformat': '.0f','showgrid': True,'gridcolor':'white'},
        xaxis_tickformat='%b %Y',
        title='7-Day averaged number of cases',
        title_x=0.03,
        title_y=0.96,
        plot_bgcolor='#E5EDF3',
        showlegend=True,
        legend={'x': 1.39, 'y': 1, 'xanchor': 'right', 'yanchor': 'top'},
        annotations=[
            dict(
                text='Country',
                showarrow=False,
                xref='paper',
                yref='paper',
                x=1.26,
                y=1.06,
                font=dict(size=12)
            )
        ],title_font=dict(size=12)
    )

    return {'data': [trace], 'layout': layout}

# Define 'cumulative number of cases' linechart callbacks
@app.callback(
    Output('cumulative-chart', 'figure'),
    [Input('country-dropdown', 'value')]
)
def update_cumulative_chart(selected_country):
    x = epidemic_cumulative.columns
    y = epidemic_cumulative.loc[selected_country]
    
    trace = go.Scatter(
        x=x,
        y=y,
        mode='lines',
        name=selected_country,
        showlegend = True
    )
    
    layout = go.Layout(
        xaxis={'title': 'Date','showgrid': True,'gridcolor':'white'},
        yaxis={'title': 'Cumulative Number of Cases'},
        title='Cumulative Number of cases',
        title_x=0.03,
        title_y=0.96,
        margin={'l': 50, 'r': 139, 't': 90, 'b': 90},
        legend={'x': 1.39, 'y': 1,'xanchor': 'right', 'yanchor': 'top'},
          annotations=[
            dict(
                text='Country',
                showarrow=False,
                xref='paper',
                yref='paper',
                x=1.26,
                y=1.06,
                font=dict(size=12)
            )
        ],title_font=dict(size=12),
        plot_bgcolor='#E5EDF3',
        paper_bgcolor='#f9f9f9'
    )

    return {'data': [trace], 'layout': layout}

#set the text callbacks for number of cases
@app.callback(
    Output('cases-text', 'children'),
    [Input('date-dropdown', 'value'), Input('country-dropdown', 'value')]
)
def update_cases_text(date, country):
    cases = confirmed_by_country.loc[country, date]
    cases = round(cases) 
    return html.Div([
        html.P(f"The Confirmed number of cases",style={'margin-bottom': '0'}),
        html.P(f"of {country} is {cases}.", style={'margin-top': '0'})
       
    ])
#set the text callbacks for number of deaths
@app.callback(
    Output('cases-text-deaths', 'children'),
    [Input('date-dropdown', 'value'), Input('country-dropdown', 'value')]
)
def update_cases_text_death(date, country):
    deaths = deaths_by_country.loc[country, date]
    deaths = round(deaths)
    return html.Div([
        html.P(f"{deaths}", style={'margin-top': '0'})
    ])

# Run the app
if __name__ == '__main__':
    app.run_server(debug=True)
